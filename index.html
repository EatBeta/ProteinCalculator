<script>

function calculate() {

// Get user inputs

const name = document.getElementById('name').value;

const weight = parseFloat(document.getElementById('weight').value);

const activityLevel = document.getElementById('activityLevel').value;



// Constants for protein and calorie calculation

const proteinPerKg = 1.8; // Recommended protein intake per kg

const caloriesPerGram = 4; // Calories per gram of protein



// Calculate protein requirement and calories

const proteinRequirement = weight * proteinPerKg;

const caloriesRequirement = proteinRequirement * caloriesPerGram;



// Display the result

const resultDiv = document.getElementById('result');

resultDiv.innerHTML = `

<p>Hello, ${name}!</p>

<p>Your daily protein requirement is approximately ${proteinRequirement.toFixed(2)} grams.</p>

<p>This corresponds to ${caloriesRequirement.toFixed(2)} calories.</p>

<p>For palm servings, consider:</p>

<ul>

<li>Small palm: ${proteinRequirement.toFixed(2)} servings</li>

<li>Medium palm: ${(proteinRequirement * 1.5).toFixed(2)} servings</li>

<li>Large palm: ${(proteinRequirement * 2).toFixed(2)} servings</li>

</ul>

`;

}

</script>
